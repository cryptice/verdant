# Weather Integration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Deliver a weather integration subsystem that powers frost/weather alerts, GDD-based harvest forecasting, weather-overlay analytics, and irrigation-aware task generation.

**Architecture:** SMHI-sourced daily weather stored in `daily_weather`, refreshed on two cadences (forecast every 6h, actuals daily at 2am). FCM push notifications for alerts. Cached GDD accumulation per `(garden, species, season)`. Design spec: `docs/plans/2026-04-18-weather-integration-design.md`.

**Tech Stack:** Quarkus + Kotlin (backend), Flyway (migrations), React + TypeScript + Tailwind (web), Kotlin + Jetpack Compose (Android), PostgreSQL, Firebase Admin SDK (backend), Firebase Messaging (Android), SMHI OpenData (external).

**Important notes:**
- All new tables use `BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY`.
- Every enum column is `VARCHAR` holding the Kotlin enum name uppercase (matches existing convention, see `Channel`, `PlantType`).
- SMHI endpoints require no auth. Point-outside-Sweden responses are empty; the ingestion service logs and skips.
- Firebase project and `google-services.json` must be provisioned by the user ã the plan cannot generate these. M2 Task 2.2 is the blocker step.
- Commits go to `main`. No backwards-compatibility shims ã no prod data exists.

```sql
-- ACTUAL arrives the morning after the day it describes.
-- FORECAST is overwritten each refresh; ACTUAL sits beside the final FORECAST for error studies.

CREATE TABLE daily_weather (

    garden_id           BIGINT       NOT NULL REFERENCES garden(id) ON DELETE CASCADE,
    date                DATE         NOT NULL,
    observation_type    VARCHAR(16)  NOT NULL CHECK (observation_type IN ('ACTUAL', 'FORECAST')),
    temp_min_c          DOUBLE PRECISION,
    temp_max_c          DOUBLE PRECISION,
    temp_mean_c         DOUBLE PRECISION,
    precipitation_mm    DOUBLE PRECISION,
    wind_max_ms         DOUBLE PRECISION,
    humidity_pct        DOUBLE PRECISION,
    fetched_at          TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    CONSTRAINT uq_daily_weather_garden_date_type UNIQUE (garden_id, date, observation_type)
);
CREATE INDEX idx_daily_weather_garden_date ON daily_weather(garden_id, date);

-- Per-garden status of the 3-year actuals backfill. NULL means not yet attempted.
ALTER TABLE garden ADD COLUMN weather_backfill_status VARCHAR(16);
```

```kotlin
package app.verdant.service.weather

import app.verdant.entity.DailyWeather
import app.verdant.entity.ObservationType
import jakarta.enterprise.context.ApplicationScoped
import org.eclipse.microprofile.config.inject.ConfigProperty
import java.net.URI
import java.net.http.HttpClient
import java.net.http.HttpRequest
import java.net.http.HttpResponse
import java.time.Duration
import java.time.Instant
import java.time.LocalDate
import java.time.ZoneId

@ApplicationScoped
class SmhiClient(
    @ConfigProperty(name = "smhi.forecast.base-url",
        defaultValue = "https://opendata-download-metfcst.smhi.se")
    private val forecastBaseUrl: String,

    @ConfigProperty(name = "smhi.archive.base-url",
        defaultValue = "https://opendata-download-metobs.smhi.se")
    private val archiveBaseUrl: String,
) {
    private val http = HttpClient.newBuilder().connectTimeout(Duration.ofSeconds(10)).build()

    fun fetchForecast(lat: Double, lon: Double, gardenId: Long): List<DailyWeather> {
        val url = "$forecastBaseUrl/api/category/pmp3g/version/2/geotype/point/" +
            "lon/${"%.4f".format(lon)}/lat/${"%.4f".format(lat)}/data.json"
        val body = get(url) ?: return emptyList()
        return SmhiForecastParser.parse(body, gardenId)
    }

    fun fetchActual(lat: Double, lon: Double, gardenId: Long, date: LocalDate): DailyWeather? {
        // Archive uses station-based URLs; the parser picks the nearest available station by param.
        val url = "$archiveBaseUrl/api/version/1.0/parameter/2/station-set/all/period/latest-day/data.json"
        val body = get(url) ?: return null
        return SmhiArchiveParser.parseForPoint(body, gardenId, date, lat, lon)
    }

    private fun get(url: String): String? {
        val req = HttpRequest.newBuilder(URI.create(url))
            .header("Accept", "application/json")
            .timeout(Duration.ofSeconds(15))
            .GET().build()
        val res = http.send(req, HttpResponse.BodyHandlers.ofString())
        if (res.statusCode() in 200..299) return res.body()
        if (res.statusCode() == 429 || res.statusCode() >= 500) return null
        error("SMHI unexpected status ${res.statusCode()} for $url")
    }
}
```

```kotlin
package app.verdant.service.weather

import app.verdant.entity.ObservationType
import app.verdant.repository.DailyWeatherRepository
import app.verdant.repository.GardenRepository
import io.quarkus.scheduler.Scheduled
import io.quarkus.runtime.annotations.RegisterForReflection
import jakarta.enterprise.context.ApplicationScoped
import jakarta.inject.Inject
import org.slf4j.LoggerFactory
import java.time.LocalDate

@ApplicationScoped
class WeatherIngestionService @Inject constructor(
    private val gardens: GardenRepository,
    private val weather: DailyWeatherRepository,
    private val smhi: SmhiClient,
    private val alerts: AlertEvaluator,   // Added in M3; inject as nullable or guard with bean-lookup until then.
) {
    private val log = LoggerFactory.getLogger(javaClass)

    @Scheduled(cron = "0 0 */6 * * ?")
    fun refreshForecasts() {
        val all = gardens.findAllWithCoordinates()
        log.info("Weather forecast refresh starting for ${all.size} gardens")
        for (g in all) {
            runCatching {
                val rows = smhi.fetchForecast(g.latitude!!, g.longitude!!, g.id!!)
                if (rows.isNotEmpty()) {
                    weather.upsert(rows)
                    alerts.evaluate(g.id)
                }
                Thread.sleep(250)
            }.onFailure { log.warn("Forecast refresh failed for garden ${g.id}", it) }
        }
    }

    @Scheduled(cron = "0 0 2 * * ?")
    fun refreshActuals() {
        val yesterday = LocalDate.now().minusDays(1)
        val all = gardens.findAllWithCoordinates()
        log.info("Weather actuals refresh starting for ${all.size} gardens (date=$yesterday)")
        for (g in all) {
            runCatching {
                val row = smhi.fetchActual(g.latitude!!, g.longitude!!, g.id!!, yesterday) ?: return@runCatching
                weather.upsert(listOf(row))
                Thread.sleep(250)
            }.onFailure { log.warn("Actuals refresh failed for garden ${g.id}", it) }
        }
    }

    fun backfillForGarden(gardenId: Long) {
        val g = gardens.findById(gardenId) ?: return
        val lat = g.latitude ?: return
        val lon = g.longitude ?: return
        gardens.setBackfillStatus(gardenId, "RUNNING")
        runCatching {
            val start = LocalDate.now().minusYears(3)
            val end = LocalDate.now().minusDays(1)
            val rows = mutableListOf<app.verdant.entity.DailyWeather>()
            var d = start
            while (!d.isAfter(end)) {
                val row = smhi.fetchActual(lat, lon, gardenId, d)
                if (row != null) rows += row
                if (rows.size >= 500) { weather.upsert(rows); rows.clear() }
                d = d.plusDays(1)
                Thread.sleep(25)
            }
            if (rows.isNotEmpty()) weather.upsert(rows)
            gardens.setBackfillStatus(gardenId, "DONE")
        }.onFailure {
            log.error("Backfill failed for garden $gardenId", it)
            gardens.setBackfillStatus(gardenId, "FAILED")
        }
    }
}
```

```kotlin
@Path("/api/weather")
@Produces(MediaType.APPLICATION_JSON)
@Authenticated
class WeatherResource @Inject constructor(
    private val weather: DailyWeatherRepository,
    private val gardens: GardenRepository,
    private val identity: SecurityIdentity,
) {
    @GET @Path("/garden/{id}")
    fun forGarden(@PathParam("id") gardenId: Long, @QueryParam("days") days: Int?): GardenWeatherResponse {
        val userId = identity.userIdLong()
        val garden = gardens.findByIdAuthorized(gardenId, userId)
            ?: throw NotFoundException("Garden not found")
        val window = (days ?: 10).coerceIn(1, 30)
        val rows = weather.findByGarden(gardenId,
            fromDate = LocalDate.now().minusDays(1),
            toDate = LocalDate.now().plusDays(window.toLong()))
        return GardenWeatherResponse(
            gardenId = gardenId,
            backfillStatus = garden.weatherBackfillStatus,
            days = rows.map { it.toResponse() },
        )
    }
}
```

```kotlin
package app.verdant.service.weather

import com.google.auth.oauth2.GoogleCredentials
import com.google.firebase.FirebaseApp
import com.google.firebase.FirebaseOptions
import com.google.firebase.messaging.FirebaseMessaging
import com.google.firebase.messaging.Message
import com.google.firebase.messaging.Notification
import io.quarkus.runtime.Startup
import jakarta.annotation.PostConstruct
import jakarta.enterprise.context.ApplicationScoped
import org.eclipse.microprofile.config.inject.ConfigProperty
import org.slf4j.LoggerFactory
import java.io.FileInputStream

@ApplicationScoped
@Startup
class FcmService(
    @ConfigProperty(name = "firebase.credentials.path", defaultValue = "")
    private val credentialsPath: String,
) {
    private val log = LoggerFactory.getLogger(javaClass)
    private var initialized = false

    @PostConstruct
    fun init() {
        if (credentialsPath.isBlank()) {
            log.warn("FIREBASE_CREDENTIALS_PATH not set ã push disabled in this environment")
            return
        }
        if (FirebaseApp.getApps().isEmpty()) {
            val opts = FirebaseOptions.builder()
                .setCredentials(GoogleCredentials.fromStream(FileInputStream(credentialsPath)))
                .build()
            FirebaseApp.initializeApp(opts)
        }
        initialized = true
    }

    data class Push(
        val title: String,
        val body: String,
        val data: Map<String, String> = emptyMap(),
    )

    /** Returns the FCM message id on success, null on "token invalid" so callers can prune. */
    fun send(token: String, push: Push): String? {
        if (!initialized) { log.info("FCM not initialized, skipping send"); return null }
        val msg = Message.builder()
            .setToken(token)
            .setNotification(Notification.builder().setTitle(push.title).setBody(push.body).build())
            .putAllData(push.data)
            .build()
        return try {
            FirebaseMessaging.getInstance().send(msg)
        } catch (e: com.google.firebase.messaging.FirebaseMessagingException) {
            if (e.messagingErrorCode?.name in setOf("UNREGISTERED", "INVALID_ARGUMENT")) return null
            log.error("FCM send failed", e)
            null
        }
    }
}
```

```kotlin
@ApplicationScoped
class AlertEvaluator @Inject constructor(
    private val weather: DailyWeatherRepository,
    private val alerts: WeatherAlertRepository,
    private val prefs: UserAlertPreferenceRepository,
    private val gardens: GardenRepository,
    private val orgs: OrganizationRepository,
    private val devices: UserDeviceRepository,
    private val fcm: FcmService,
) {
    private val defaults = mapOf(
        ThresholdType.FROST to -0.0,
        ThresholdType.HEAT to 30.0,
        ThresholdType.HIGH_WIND to 15.0,
        ThresholdType.HEAVY_RAIN to 30.0,
    )

    /** Called after a forecast refresh for a single garden. */
    fun evaluate(gardenId: Long) {
        val garden = gardens.findById(gardenId) ?: return
        val window = LocalDate.now()..LocalDate.now().plusDays(3)
        val forecast = weather.findByGarden(gardenId, window.start, window.endInclusive)
            .filter { it.observationType == ObservationType.FORECAST }
        val recipients = orgs.findUsersInOrg(garden.orgId)

        for (row in forecast) {
            for ((type, defaultValue) in defaults) {
                if (type == ThresholdType.FROST && !hasActivePlantsOutdoors(gardenId)) continue
                if (type != ThresholdType.FROST && !hasActiveSeason(gardenId)) continue

                val tripped = tripped(row, type, defaultValue)
                if (!tripped.first) continue
                val observedValue = tripped.second

                for (userId in recipients) {
                    val pref = prefs.getOrDefault(userId, type)
                    if (!pref.enabled) continue
                    val threshold = pref.customValue ?: defaultValue
                    if (!trippedValue(observedValue, type, threshold)) continue
                    if (alerts.existsFor(gardenId, row.date, type, userId)) continue

                    val alert = WeatherAlert(
                        gardenId = gardenId, targetDate = row.date,
                        thresholdType = type, thresholdValue = observedValue,
                        fcmMessageId = null, userId = userId)

                    val sent = sendTo(userId, gardenId, garden.name, alert)
                    alerts.insert(alert.copy(fcmMessageId = sent))
                }
            }
        }
    }

    private fun tripped(row: DailyWeather, type: ThresholdType, def: Double): Pair<Boolean, Double> =
        when (type) {
            ThresholdType.FROST -> (row.tempMinC != null && row.tempMinC <= def) to (row.tempMinC ?: 0.0)
            ThresholdType.HEAT -> (row.tempMaxC != null && row.tempMaxC >= def) to (row.tempMaxC ?: 0.0)
            ThresholdType.HIGH_WIND -> (row.windMaxMs != null && row.windMaxMs >= def) to (row.windMaxMs ?: 0.0)
            ThresholdType.HEAVY_RAIN -> (row.precipitationMm != null && row.precipitationMm >= def) to (row.precipitationMm ?: 0.0)
            else -> false to 0.0
        }

    private fun trippedValue(value: Double, type: ThresholdType, custom: Double): Boolean =
        when (type) {
            ThresholdType.FROST -> value <= custom
            ThresholdType.HEAT, ThresholdType.HIGH_WIND, ThresholdType.HEAVY_RAIN -> value >= custom
            else -> false
        }

    private fun sendTo(userId: Long, gardenId: Long, gardenName: String, alert: WeatherAlert): String? {
        val tokens = devices.findByUser(userId)
        val push = FcmService.Push(
            title = titleFor(alert.thresholdType, gardenName),
            body = bodyFor(alert),
            data = mapOf(
                "type" to "weather_alert",
                "gardenId" to gardenId.toString(),
                "alertType" to alert.thresholdType.name,
                "targetDate" to alert.targetDate.toString()))
        for (t in tokens) {
            val id = fcm.send(t.fcmToken, push)
            if (id == null) devices.deleteByToken(t.fcmToken) else return id
        }
        return null
    }

    private fun titleFor(t: ThresholdType, garden: String): String = when (t) {
        ThresholdType.FROST -> "Frostvarning ã $garden"
        ThresholdType.HEAT -> "VûÊrmevarning ã $garden"
        ThresholdType.HIGH_WIND -> "BlûËstvarning ã $garden"
        ThresholdType.HEAVY_RAIN -> "Regnvarning ã $garden"
        else -> "VûÊder ã $garden"
    }

    private fun bodyFor(a: WeatherAlert): String = when (a.thresholdType) {
        ThresholdType.FROST -> "Ner mot ${"%.0f".format(a.thresholdValue)}ô¯C natten till ${a.targetDate}"
        ThresholdType.HEAT -> "Upp mot ${"%.0f".format(a.thresholdValue)}ô¯C ${a.targetDate}"
        ThresholdType.HIGH_WIND -> "Vindar upp mot ${"%.0f".format(a.thresholdValue)} m/s ${a.targetDate}"
        ThresholdType.HEAVY_RAIN -> "NederbûÑrd ca ${"%.0f".format(a.thresholdValue)} mm ${a.targetDate}"
        else -> "VûÊder ${a.targetDate}"
    }

    private fun hasActivePlantsOutdoors(gardenId: Long): Boolean = gardens.hasActivePlantsOutdoors(gardenId)
    private fun hasActiveSeason(gardenId: Long): Boolean = gardens.hasActiveSeason(gardenId)
}
```

Key methods:
- `refreshGarden(gardenId: Long)` ã called from the actuals cron. For each `(species, season)` with plant events in the garden, recompute accumulated GDD from the `as_of_date` forward using ACTUAL rows.
- `forPlant(plantId: Long): PlantGddResponse` ã walks cached actuals + forecast. Returns `accumulatedGddToday`, `predictedFirstBloomDate`, `predictedHarvestDate`, `confidence`.

```kotlin
data class PlantGddResponse(
    val accumulatedGddToday: Double,
    val predictedFirstBloomDate: LocalDate?,
    val predictedHarvestDate: LocalDate?,
    val confidence: String, // "high" | "forecast-limited" | "low"
)

fun forPlant(plantId: Long): PlantGddResponse {
    val plant = plants.findById(plantId) ?: error("not found")
    val species = species.findById(plant.speciesId) ?: error("species missing")
    val base = species.gddBaseTempC ?: 10.0
    val seasonId = plant.seasonId ?: return fallbackByDays(plant, species)
    val sowDate = plantEvents.findSown(plantId)?.occurredAt?.toLocalDate() ?: return fallbackByDays(plant, species)

    val actuals = weather.findActualsByGarden(plant.gardenId, sowDate, LocalDate.now())
    val forecast = weather.findByGarden(plant.gardenId, LocalDate.now(), LocalDate.now().plusDays(10))
        .filter { it.observationType == ObservationType.FORECAST }

    var gdd = 0.0
    var firstBloom: LocalDate? = null
    var harvest: LocalDate? = null
    val bloomAt = species.gddToFirstBloom
    val harvestAt = species.gddToHarvest

    fun step(row: DailyWeather) {
        val t = row.tempMeanC ?: return
        gdd += (t - base).coerceAtLeast(0.0)
        if (firstBloom == null && bloomAt != null && gdd >= bloomAt) firstBloom = row.date
        if (harvest == null && harvestAt != null && gdd >= harvestAt) harvest = row.date
    }

    (actuals + forecast).sortedBy { it.date }.forEach(::step)

    val today = actuals.lastOrNull()?.date ?: sowDate
    val confidence = when {
        species.gddBaseTempC == null || harvestAt == null -> "low"
        harvest == null -> "forecast-limited"
        else -> "high"
    }

    val predictedFirstBloom = firstBloom ?: fallbackDate(plant, species, span = bloomAt)
    val predictedHarvest = harvest ?: fallbackDate(plant, species, span = harvestAt)

    return PlantGddResponse(
        accumulatedGddToday = gdd,
        predictedFirstBloomDate = predictedFirstBloom,
        predictedHarvestDate = predictedHarvest,
        confidence = confidence,
    )
}

private fun fallbackByDays(plant: Plant, sp: Species): PlantGddResponse {
    val d = sp.daysToHarvest ?: 90
    val sow = plantEvents.findSown(plant.id!!)?.occurredAt?.toLocalDate() ?: LocalDate.now()
    return PlantGddResponse(
        accumulatedGddToday = 0.0,
        predictedFirstBloomDate = null,
        predictedHarvestDate = sow.plusDays(d.toLong()),
        confidence = "low",
    )
}

private fun fallbackDate(plant: Plant, sp: Species, span: Double?): LocalDate? {
    if (span == null) return null
    val d = sp.daysToHarvest ?: 90
    val sow = plantEvents.findSown(plant.id!!)?.occurredAt?.toLocalDate() ?: LocalDate.now()
    return sow.plusDays(d.toLong())
}
```
