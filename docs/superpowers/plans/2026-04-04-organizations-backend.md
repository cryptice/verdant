# Organizations Backend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add organization ownership to all data, replacing per-user ownership, with org CRUD, membership, invites, and a central org context filter.

**Architecture:** New `organization` and `org_member` tables own all operational data. A V7 migration replaces `user_id`/`owner_id` with `org_id` across 17 tables. An `OrgContextFilter` reads `X-Organization-Id` from request headers, validates membership, and injects the org ID into the JAX-RS request context. All repositories, services, and resources are updated to use `orgId` instead of `userId`.

**Tech Stack:** Quarkus 3.17, Kotlin 2.1, raw JDBC with AgroalDataSource, JWT auth

```kotlin
package app.verdant.repository

import app.verdant.entity.Organization
import io.agroal.api.AgroalDataSource
import jakarta.enterprise.context.ApplicationScoped
import java.sql.ResultSet
import java.sql.Statement

@ApplicationScoped
class OrganizationRepository(private val ds: AgroalDataSource) {

    fun findById(id: Long): Organization? =
        ds.connection.use { conn ->
            conn.prepareStatement("SELECT * FROM organization WHERE id = ?").use { ps ->
                ps.setLong(1, id)
                ps.executeQuery().use { rs -> if (rs.next()) rs.toOrganization() else null }
            }
        }

    fun persist(org: Organization): Organization {
        ds.connection.use { conn ->
            conn.prepareStatement(
                "INSERT INTO organization (name, emoji, created_at, updated_at) VALUES (?, ?, now(), now())",
                Statement.RETURN_GENERATED_KEYS
            ).use { ps ->
                ps.setString(1, org.name)
                ps.setString(2, org.emoji)
                ps.executeUpdate()
                ps.generatedKeys.use { rs ->
                    rs.next()
                    return org.copy(id = rs.getLong(1))
                }
            }
        }
    }

    fun update(org: Organization) {
        ds.connection.use { conn ->
            conn.prepareStatement(
                "UPDATE organization SET name = ?, emoji = ?, updated_at = now() WHERE id = ?"
            ).use { ps ->
                ps.setString(1, org.name)
                ps.setString(2, org.emoji)
                ps.setLong(3, org.id!!)
                ps.executeUpdate()
            }
        }
    }

    fun delete(id: Long) {
        ds.connection.use { conn ->
            conn.prepareStatement("DELETE FROM organization WHERE id = ?").use { ps ->
                ps.setLong(1, id)
                val rows = ps.executeUpdate()
                if (rows == 0) throw jakarta.ws.rs.NotFoundException("Organization not found")
            }
        }
    }

    private fun ResultSet.toOrganization() = Organization(
        id = getLong("id"),
        name = getString("name"),
        emoji = getString("emoji"),
        createdAt = getTimestamp("created_at").toInstant(),
        updatedAt = getTimestamp("updated_at").toInstant(),
    )
}
```

```kotlin
package app.verdant.filter

import app.verdant.repository.OrgMemberRepository
import jakarta.annotation.Priority
import jakarta.ws.rs.Priorities
import jakarta.ws.rs.container.ContainerRequestContext
import jakarta.ws.rs.container.ContainerRequestFilter
import jakarta.ws.rs.core.Response
import jakarta.ws.rs.ext.Provider
import org.eclipse.microprofile.jwt.JsonWebToken

@Provider
@Priority(Priorities.AUTHORIZATION + 1)
class OrgContextFilter(
    private val jwt: JsonWebToken,
    private val orgMemberRepository: OrgMemberRepository,
    private val orgContext: OrgContext,
) : ContainerRequestFilter {

    // Paths that don't require an org context
    private val exemptPrefixes = listOf(
        "/api/auth/",
        "/api/users/me",
        "/api/organizations",
        "/api/invites",
        "/api/admin/",
        "/api/dev/",
        "/api/openapi",
        "/api/swagger-ui",
        "/q/",
    )

    override fun filter(context: ContainerRequestContext) {
        val path = context.uriInfo.path
        if (exemptPrefixes.any { path.startsWith(it) }) return
        if (jwt.subject == null) return // Not authenticated, let @Authenticated handle it

        val userId = jwt.subject.toLong()
        orgContext.userId = userId

        val orgIdHeader = context.getHeaderString("X-Organization-Id")
        if (orgIdHeader == null) {
            context.abortWith(
                Response.status(Response.Status.BAD_REQUEST)
                    .entity(mapOf("message" to "X-Organization-Id header is required"))
                    .build()
            )
            return
        }

        val orgId = orgIdHeader.toLongOrNull()
        if (orgId == null) {
            context.abortWith(
                Response.status(Response.Status.BAD_REQUEST)
                    .entity(mapOf("message" to "Invalid X-Organization-Id"))
                    .build()
            )
            return
        }

        if (!orgMemberRepository.isMember(orgId, userId)) {
            context.abortWith(
                Response.status(Response.Status.FORBIDDEN)
                    .entity(mapOf("message" to "Not a member of this organization"))
                    .build()
            )
            return
        }

        orgContext.orgId = orgId
    }
}
```

```kotlin
package app.verdant.service

import app.verdant.dto.*
import app.verdant.entity.*
import app.verdant.repository.*
import jakarta.enterprise.context.ApplicationScoped
import jakarta.transaction.Transactional
import jakarta.ws.rs.BadRequestException
import jakarta.ws.rs.ForbiddenException
import jakarta.ws.rs.NotFoundException

@ApplicationScoped
class OrganizationService(
    private val orgRepo: OrganizationRepository,
    private val memberRepo: OrgMemberRepository,
    private val inviteRepo: OrgInviteRepository,
    private val userRepo: UserRepository,
) {
    fun getOrganizationsForUser(userId: Long): List<OrganizationResponse> {
        val memberships = memberRepo.findByUserId(userId)
        return memberships.mapNotNull { m ->
            val org = orgRepo.findById(m.orgId) ?: return@mapNotNull null
            OrganizationResponse(
                id = org.id!!, name = org.name, emoji = org.emoji,
                role = m.role, createdAt = org.createdAt,
            )
        }
    }

    @Transactional
    fun createOrganization(request: CreateOrganizationRequest, userId: Long): OrganizationResponse {
        val org = orgRepo.persist(Organization(name = request.name, emoji = request.emoji))
        memberRepo.persist(OrgMember(orgId = org.id!!, userId = userId, role = OrgRole.OWNER))
        return OrganizationResponse(
            id = org.id!!, name = org.name, emoji = org.emoji,
            role = OrgRole.OWNER, createdAt = org.createdAt,
        )
    }

    fun updateOrganization(orgId: Long, request: UpdateOrganizationRequest, userId: Long): OrganizationResponse {
        val org = orgRepo.findById(orgId) ?: throw NotFoundException("Organization not found")
        val member = memberRepo.findByOrgAndUser(orgId, userId) ?: throw ForbiddenException()
        if (member.role != OrgRole.OWNER) throw ForbiddenException("Only owners can update the organization")
        val updated = org.copy(
            name = request.name ?: org.name,
            emoji = request.emoji ?: org.emoji,
        )
        orgRepo.update(updated)
        return OrganizationResponse(
            id = updated.id!!, name = updated.name, emoji = updated.emoji,
            role = member.role, createdAt = updated.createdAt,
        )
    }

    fun deleteOrganization(orgId: Long, userId: Long) {
        val member = memberRepo.findByOrgAndUser(orgId, userId) ?: throw ForbiddenException()
        if (member.role != OrgRole.OWNER) throw ForbiddenException("Only owners can delete the organization")
        orgRepo.delete(orgId)
    }

    fun getMembers(orgId: Long): List<OrgMemberResponse> {
        val members = memberRepo.findByOrgId(orgId)
        return members.mapNotNull { m ->
            val user = userRepo.findById(m.userId) ?: return@mapNotNull null
            OrgMemberResponse(
                id = m.id!!, userId = user.id!!, email = user.email,
                displayName = user.displayName, avatarUrl = user.avatarUrl,
                role = m.role, joinedAt = m.joinedAt,
            )
        }
    }

    fun inviteMember(orgId: Long, request: InviteMemberRequest, userId: Long): OrgInviteResponse {
        val member = memberRepo.findByOrgAndUser(orgId, userId) ?: throw ForbiddenException()
        if (member.role != OrgRole.OWNER) throw ForbiddenException("Only owners can invite members")
        val inviter = userRepo.findById(userId) ?: throw NotFoundException("User not found")
        val org = orgRepo.findById(orgId) ?: throw NotFoundException("Organization not found")
        val invite = inviteRepo.persist(OrgInvite(orgId = orgId, email = request.email, invitedBy = userId))
        return OrgInviteResponse(
            id = invite.id!!, orgId = orgId, orgName = org.name,
            email = invite.email, invitedByName = inviter.displayName,
            status = invite.status, createdAt = invite.createdAt,
        )
    }

    fun removeMember(orgId: Long, targetUserId: Long, userId: Long) {
        val member = memberRepo.findByOrgAndUser(orgId, userId) ?: throw ForbiddenException()
        if (member.role != OrgRole.OWNER && userId != targetUserId) throw ForbiddenException("Only owners can remove members")
        val target = memberRepo.findByOrgAndUser(orgId, targetUserId) ?: throw NotFoundException("Member not found")
        if (target.role == OrgRole.OWNER && userId == targetUserId) throw BadRequestException("Cannot remove yourself as the only owner")
        memberRepo.delete(orgId, targetUserId)
    }

    fun getPendingInvitesForUser(email: String): List<OrgInviteResponse> {
        return inviteRepo.findPendingByEmail(email).mapNotNull { invite ->
            val org = orgRepo.findById(invite.orgId) ?: return@mapNotNull null
            val inviter = userRepo.findById(invite.invitedBy) ?: return@mapNotNull null
            OrgInviteResponse(
                id = invite.id!!, orgId = invite.orgId, orgName = org.name,
                email = invite.email, invitedByName = inviter.displayName,
                status = invite.status, createdAt = invite.createdAt,
            )
        }
    }

    @Transactional
    fun acceptInvite(inviteId: Long, userId: Long): OrganizationResponse {
        val invite = inviteRepo.findById(inviteId) ?: throw NotFoundException("Invite not found")
        val user = userRepo.findById(userId) ?: throw NotFoundException("User not found")
        if (invite.email != user.email) throw ForbiddenException("This invite is not for you")
        if (invite.status != InviteStatus.PENDING) throw BadRequestException("Invite is no longer pending")
        inviteRepo.updateStatus(inviteId, InviteStatus.ACCEPTED)
        memberRepo.persist(OrgMember(orgId = invite.orgId, userId = userId, role = OrgRole.MEMBER))
        val org = orgRepo.findById(invite.orgId)!!
        return OrganizationResponse(
            id = org.id!!, name = org.name, emoji = org.emoji,
            role = OrgRole.MEMBER, createdAt = org.createdAt,
        )
    }

    fun declineInvite(inviteId: Long, userId: Long) {
        val invite = inviteRepo.findById(inviteId) ?: throw NotFoundException("Invite not found")
        val user = userRepo.findById(userId) ?: throw NotFoundException("User not found")
        if (invite.email != user.email) throw ForbiddenException("This invite is not for you")
        if (invite.status != InviteStatus.PENDING) throw BadRequestException("Invite is no longer pending")
        inviteRepo.updateStatus(inviteId, InviteStatus.DECLINED)
    }
}
```

```kotlin
package app.verdant.resource

import app.verdant.dto.*

import jakarta.validation.Valid
import jakarta.ws.rs.*

import org.eclipse.microprofile.jwt.JsonWebToken

@Path("/api/organizations")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Authenticated
class OrganizationResource(
    private val orgService: OrganizationService,
    private val jwt: JsonWebToken,
) {
    private fun userId() = jwt.subject.toLong()

    @GET
    fun list() = orgService.getOrganizationsForUser(userId())

    @POST
    fun create(@Valid request: CreateOrganizationRequest): Response {
        val org = orgService.createOrganization(request, userId())
        return Response.status(Response.Status.CREATED).entity(org).build()
    }

    @PUT
    @Path("/{id}")
    fun update(@PathParam("id") id: Long, @Valid request: UpdateOrganizationRequest) =
        orgService.updateOrganization(id, request, userId())

    @DELETE
    @Path("/{id}")
    fun delete(@PathParam("id") id: Long): Response {
        orgService.deleteOrganization(id, userId())
        return Response.noContent().build()
    }

    @GET
    @Path("/{id}/members")
    fun listMembers(@PathParam("id") id: Long) = orgService.getMembers(id)

    @POST
    @Path("/{id}/invite")
    fun invite(@PathParam("id") id: Long, @Valid request: InviteMemberRequest): Response {
        val invite = orgService.inviteMember(id, request, userId())
        return Response.status(Response.Status.CREATED).entity(invite).build()
    }

    @DELETE
    @Path("/{id}/members/{userId}")
    fun removeMember(@PathParam("id") id: Long, @PathParam("userId") targetUserId: Long): Response {
        orgService.removeMember(id, targetUserId, userId())
        return Response.noContent().build()
    }
}

    }
}
```
